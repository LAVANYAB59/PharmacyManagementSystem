Pharmacy Table:

CREATE TABLE PHARMACY_TB(
RECORD_ID VARCHAR2(12 BYTE) PRIMARY KEY,
MEDICINENAME VARCHAR2(40 BYTE) NOT NULL,
SUPPLIERNAME VARCHAR2(40 BYTE) NOT NULL,
PURCHASE_DATE  DATE NOT NULL,
QUANTITY NUMBER,
PRICE NUMBER,
REMARKS VARCHAR2(100 BYTE)
);

SELECT *FROM PHARMACY_TB;

Sequence Table:

CREATE SEQUENCE PHARMACY_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
MINVALUE 1
NOCACHE
NOCYCLE;
